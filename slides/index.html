Why do we test?

  • Validate that our code follows behavior that our clients expect
  • Prevent regression when code changes

The underlying goal is working software.

What are ways we can test?

Manual Testing

  • Testing through the UI
  • Developer testing in an interactive environment (i.e. REPL)

Automated Testing

  • Unit Tests
  • Integration Tests
  • Functional Tests/End to End Tests (i.e. Selenium)

JUnit Example


							@Test
							public void canAddItems() {
							    //given
							    ShoppingCart cart = new ShoppingCart();
							    Product nomProduct = new Product("A71243E2", "Num Noms Series 2 Sparkle Cupcake Playset", new BigDecimal(9.59));
							    Product ponyProduct = new Product("4459EAD4", "My Little Pony Pinkie Pie Sweet Style Pony Playset",
							            new BigDecimal(21.99));
							    BigDecimal expectedSubtotal = new BigDecimal(31.58).setScale(2, BigDecimal.ROUND_HALF_UP);
							    LineItem firstExpectedItem = new LineItem(nomProduct, 1);
							    LineItem secondExpectedItem = new LineItem(ponyProduct, 1);

							    //when
							    cart.add(nomProduct);
							    cart.add(ponyProduct);

							    //then
							    Assert.assertEquals(2, cart.getItemCount());
							    Assert.assertEquals(expectedSubtotal, cart.getSubtotal());
							    assertLineEntry(firstExpectedItem, cart.getLineItems().get(0));
							    assertLineEntry(secondExpectedItem, cart.getLineItems().get(1));
							}

							private void assertLineEntry(final LineItem expectedItem, final LineItem actualItem) {
							    Assert.assertEquals(expectedItem.getSku(), actualItem.getSku());
							    Assert.assertEquals(expectedItem.getName(), actualItem.getName());
							    Assert.assertEquals(expectedItem.getQuantity(), actualItem.getQuantity());
							    Assert.assertEquals(expectedItem.getPrice(), actualItem.getPrice());
							}

Fake

Have working implementations but take shortcuts that make them not suitable for production.


							private static class FakeSalesTaxDeterminer implements SalesTaxDeterminer {
							    private final Map<String, Float> salesTaxMap;

							    public FakeSalesTaxDeterminer() {
							        salesTaxMap = new HashMap<String, Float>();
							        salesTaxMap.put("TN", 0.0925f);
							    }

							    @Override
							    public float determineSalesTax(final String state) {
							        return salesTaxMap.getOrDefault(state, 0.01f);
							    }
							}

							@Before
							public void setUp() {
							    //....omitted
							    salesTaxDeterminer = new FakeSalesTaxDeterminer();
							    orderRegister = new OrderRegister(orderRepository, salesTaxDeterminer);
							    //....omitted
							}

							@Test
							public void canPlaceOrder() {
							    //given
							    //expectedSubtotal = $10
							    //expectedTax = $0.93

							    //when
							    Order order = orderRegister.placeOrder(cart, "daniel.j.aquino@gmail.com", shippingAddress);

							    //then
							    //....omitted
							    Assert.assertEquals(expectedTax, new BigDecimal(0.93).setScale(2, BigDecimal.ROUND_HALF_UP));
							}

Stub

Provide canned responses when called


							private static class StubSalesTaxDeterminer implements SalesTaxDeterminer {
					        private final float taxPercentage;

					        public StubSalesTaxDeterminer(final float taxPercentage) {
					            this.taxPercentage = taxPercentage;
					        }

					        @Override
					        public float determineSalesTax(final String state) {
					            return taxPercentage;
					        }
					    }

							@Before
							public void setUp() {
							    //....omitted
							    salesTaxDeterminer = new FakeSalesTaxDeterminer(0.0925);
							    orderRegister = new OrderRegister(orderRepository, salesTaxDeterminer);
							    //....omitted
							}

							@Test
							public void canPlaceOrder() {
							    //given
							    //expectedSubtotal = $10

							    //when
							    Order order = orderRegister.placeOrder(cart, "daniel.j.aquino@gmail.com", shippingAddress);

							    //then
							    //....omitted
							    Assert.assertEquals(expectedTax, new BigDecimal(0.93).setScale(2, BigDecimal.ROUND_HALF_UP));

Spy

Record information on how they were called.


							private class SpyOrderRepository implements OrderRepository {
							    private Order order;

							    public SpyOrderRepository() {
							    }

							    @Override
							    public void save(final Order order) {
							        this.order = order;
							    }

							    public Order getOrder() {
							        return order;
							    }
							}

							@Before
							public void setUp() {
							    //....omitted
							    orderRepository = new SpyOrderRepository();
							    orderRegister = new OrderRegister(orderRepository, salesTaxDeterminer);
							    //....omitted
							}

							@Test
							public void placedOrderIsPersisted() {
							    //given omitted

							    //when
							    orderRegister.placeOrder(cart, "daniel.j.aquino@gmail.com", shippingAddress);

							    //then
							    Order savedOrder = orderRepository.getOrder();
							    Assert.assertEquals("daniel.j.aquino@gmail.com", savedOrder.getUserId());
							}

Mock

Pre-programmed with expectations of calls they should receive


							@Before
							public void setUp() {
							    orderRepository = Mockito.mock(OrderRepository.class);
							    salesTaxDeterminer = Mockito.mock(SalesTaxDeterminer.class);
							    orderRegister = new OrderRegister(orderRepository, salesTaxDeterminer);
							}

							@Test
							public void placedOrderIsPersisted() {
							    //given
							    cart.add(nomProduct);

							    //when
							    orderRegister.placeOrder(cart, "daniel.j.aquino@gmail.com", shippingAddress);

							    //then
							    Mockito.verify(orderRepository).save(Mockito.any(Order.class));
							}
